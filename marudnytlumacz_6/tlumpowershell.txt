$inputPath = "en-7.2.json5"
$outputPath = "pl-7.2.json5"

$lines = Get-Content $inputPath
$output = New-Object System.Collections.Generic.List[string]

$output.Add("{")
foreach ($line in $lines) {
    $trimmed = $line.Trim()
    if ($trimmed -eq "{" -or $trimmed -eq "}" -or $trimmed -eq "") { continue }
    
    $output.Add("  // " + $trimmed)
    $output.Add("  " + $trimmed)
}
$output.Add("}")

$output | Set-Content $outputPath -Encoding UTF8
